Error definition¶

In [15]:
rmse <- function(M,P){
    # INPUTS
    #  M    : measurements
    #  P    : predictions
    
    n <- length(M)
    
    E = (M - P)
    S = E^2
    M = sum(S) / n
    R = sqrt(M)
    
    R
}
